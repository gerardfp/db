# JOINS

```sql
CREATE TABLE clientes (
    id_cliente INT,
    nombre TEXT,
    email TEXT,
    telefono TEXT,
    fecha_alta DATE
);

CREATE TABLE productos (
    id_producto INT,
    descripcion TEXT,
    precio NUMERIC
);

CREATE TABLE favoritos (
    id_cliente INT,
    id_producto INT
)

CREATE TABLE reseñas (
    id_cliente,
    id_producto,
    reseña TEXT,
    valoracion INT
)
```
